%use for chapters before the main work e.g. acknowledgments, abstracts, etc.
\newcommand{\chapterBeforeMain}[1]{
	\chapter* {#1}
	\addcontentsline{toc}{section}{#1}
	\pagestyle{chapterbeforemain}
}

%use for Table of Contents as \chapterTOC{Table of Contents}
\newcommand{\chapterTOC}[1]{
	\cleardoublepage
	\phantomsection
	\addcontentsline{toc}{section}{#1}
}

%use for List of Figures as \chapterLOF{List of Figures}
\newcommand{\chapterLOF}[1]{
	\pagestyle{chapterbeforemain}
	\newpage
	\renewcommand{\cftfigpresnum}{Figure }
	\setlength{\cftfignumwidth}{50pt}
	\cleardoublepage	
	\phantomsection
	\addcontentsline{toc}{section}{#1}
}

%use for List of Tables as \chapterLOT{List of Tables}
\newcommand{\chapterLOT}[1]{
	\pagestyle{chapterbeforemain}
	\newpage
	\renewcommand{\cfttabpresnum}{Table }
	\setlength{\cfttabnumwidth}{50pt}
	\cleardoublepage	
	\phantomsection
	\addcontentsline{toc}{section}{#1}
}

%use as \chapterCoverFirst{title of chapter}. image file: img/chapter_template.png. 
%for first chapter cover setting the numbering and header marks
\newcommand{\chapterCoverFirst}[1]{

	\renewcommand{\chaptermark}[1]{\markboth{##1}{}}
	\setcounter{page}{1}
	\pagenumbering{arabic}
	\thispagestyle{empty}
	\begin{overpic}[]%
		{img/chapter_template.png}
		\put(80,61){{\chaptercoverfont #1}}
	\end{overpic}	
	\newpage 
	\thispagestyle{empty}
	\mbox{}	
	\newpage
	\pagestyle{thesischapter}
	\fancypagestyle{plain}{%set chapter page to show page num and chapter title
		\fancyhf{}
		\fancyhf[HOL]{\thepage}
		\fancyhf[HOR]{\leftmark}
		\fancyhf[HEL]{\leftmark}
		\fancyhf[HER]{\thepage}
		\fancyhf[FEC]{}
		\fancyhf[FOC]{}
	}
}
